  • Contratti con piu' di una business entity come contracting authority - query 
    • PREFIX pc: <http://purl.org/procurement/public-contracts#>
PREFIX dcterms: <http://purl.org/dc/terms/>

SELECT ?contract (COUNT(?auth) AS ?count) {
   SELECT DISTINCT ?contract ?auth
   WHERE {
       ?contract pc:contractingAutority ?auth.
   }
}
GROUP BY ?contract HAVING (count(*)>1)
ORDER BY DESC(?count)
LIMIT 100
  • Contratti con piu' di un XML sorgente - query 
    • PREFIX dcterms: <http://purl.org/dc/terms/>

SELECT ?contract (COUNT(?source) AS ?count)
{
    SELECT DISTINCT ?contract ?source
    WHERE {
        ?contract dcterms:identifier ?CIG.
        ?contract dcterms:source ?source.
    }
}
GROUP BY ?contract HAVING (count(*)>1)
ORDER BY DESC (?count)
LIMIT 100
  • Contratti con piu' di una label - query 
    • PREFIX dcterms: <http://purl.org/dc/terms/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>

SELECT ?contract ?CIG (count(?label) as ?c)
WHERE {
    ?contract dcterms:identifier ?CIG.

}

ORDER BY DESC (?c)
LIMIT 100

  • Anni per numero di utilizzi - query 
    • PREFIX pc: <http://purl.org/procurement/public-contracts#>
PREFIX time: <http://www.w3.org/2006/time#>

SELECT distinct ?year (count(?year) as ?a)
WHERE {
    ?c a pc:Contract .
    ?c time:year ?year .
}
GROUP BY (?year)
ORDER BY DESC (?a)
